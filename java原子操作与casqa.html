// AtomicInteger实现原理解析 public class AtomicIntegerAnalysis { // 模拟AtomicInteger的核心实现 public static class MyAtomicInteger { private static final Unsafe unsafe = getUnsafe(); private static final long valueOffset; static { try { valueOffset = unsafe.objectFieldOffset (MyAtomicInteger.class.getDeclaredField("value")); } catch (Exception ex) { throw new Error(ex); } } private volatile int value; // 使用volatile保证可见性 public MyAtomicInteger(int initialValue) { value = initialValue; } // 获取当前值 public final int get() { return value; } // 设置新值 public final void set(int newValue) { value = newValue; } // CAS操作 public final boolean compareAndSet(int expect, int update) { return unsafe.compareAndSwapInt(this, valueOffset, expect, update); } // 原子递增 public final int incrementAndGet() { for (;;) { int current = get(); int next = current + 1; if (compareAndSet(current, next)) return next; // 如果CAS失败，继续循环重试 } } // 原子递减 public final int decrementAndGet() { for (;;) { int current = get(); int next = current - 1; if (compareAndSet(current, next)) return next; } } // 原子加法 public final int addAndGet(int delta) { for (;;) { int current = get(); int next = current + delta; if (compareAndSet(current, next)) return next; } } } // 性能对比测试 public static void performanceComparison() { final int THREAD_COUNT = 10; final int INCREMENT_COUNT = 100000; // 测试AtomicInteger AtomicInteger atomicCounter = new AtomicInteger(0); long startTime = System.currentTimeMillis(); Thread[] threads = new Thread[THREAD_COUNT]; for (int i = 0; i < THREAD_COUNT; i++) { threads[i] = new Thread(() -> { for (int j = 0; j < INCREMENT_COUNT; j++) { atomicCounter.incrementAndGet(); } }); } for (Thread thread : threads) { thread.start(); } for (Thread thread : threads) { try { thread.join(); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } long endTime = System.currentTimeMillis(); System.out.println("AtomicInteger结果: " + atomicCounter.get() + ", 耗时: " + (endTime - startTime) + "ms"); // 测试synchronized Counter syncCounter = new Counter(); startTime = System.currentTimeMillis(); for (int i = 0; i < THREAD_COUNT; i++) { threads[i] = new Thread(() -> { for (int j = 0; j < INCREMENT_COUNT; j++) { syncCounter.increment(); } }); } for (Thread thread : threads) { thread.start(); } for (Thread thread : threads) { try { thread.join(); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } endTime = System.currentTimeMillis(); System.out.println("Synchronized结果: " + syncCounter.getValue() + ", 耗时: " + (endTime - startTime) + "ms"); } // synchronized版本的计数器 static class Counter { private int value = 0; public synchronized void increment() { value++; } public synchronized int getValue() { return value; } } }

// ABA问题演示 public class ABAProblemDemo { // 模拟ABA问题 public static void demonstrateABAProblem() { AtomicReference<String> atomicRef = new AtomicReference<>("A"); // 线程1：模拟ABA操作 Thread thread1 = new Thread(() -> { String value = atomicRef.get(); System.out.println("Thread1 读取到值: " + value); try { // 模拟一些处理时间，在此期间其他线程修改了值 Thread.sleep(2000); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } // 尝试CAS操作 boolean success = atomicRef.compareAndSet("A", "B"); System.out.println("Thread1 CAS操作结果: " + success + ", 当前值: " + atomicRef.get()); }); // 线程2和线程3：制造ABA场景 Thread thread2 = new Thread(() -> { try { Thread.sleep(500); boolean success = atomicRef.compareAndSet("A", "C"); System.out.println("Thread2 将A改为C: " + success); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } }); Thread thread3 = new Thread(() -> { try { Thread.sleep(1000); boolean success = atomicRef.compareAndSet("C", "A"); System.out.println("Thread3 将C改回A: " + success); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } }); thread1.start(); thread2.start(); thread3.start(); try { thread1.join(); thread2.join(); thread3.join(); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } }

使用AtomicStampedReference解决ABA问题：

// 使用AtomicStampedReference解决ABA问题 public class ABASolution { public static void solveABAProblem() { // 使用AtomicStampedReference，包含值和版本号 AtomicStampedReference<String> stampedRef = new AtomicStampedReference<>("A", 1); // 线程1：带版本号的CAS操作 Thread thread1 = new Thread(() -> { int[] stampHolder = new int[1]; String value = stampedRef.get(stampHolder); int stamp = stampHolder[0]; System.out.println("Thread1 读取到值: " + value + ", 版本号: " + stamp); try { Thread.sleep(2000); // 模拟处理时间 } catch (InterruptedException e) { Thread.currentThread().interrupt(); } // 使用期望的值和版本号进行CAS boolean success = stampedRef.compareAndSet(value, "B", stamp, stamp + 1); int[] currentStampHolder = new int[1]; String currentValue = stampedRef.get(currentStampHolder); System.out.println("Thread1 CAS操作结果: " + success + ", 当前值: " + currentValue + ", 当前版本号: " + currentStampHolder[0]); }); // 线程2：修改值和版本号 Thread thread2 = new Thread(() -> { try { Thread.sleep(500); int[] stampHolder = new int[1]; String value = stampedRef.get(stampHolder); int stamp = stampHolder[0]; boolean success = stampedRef.compareAndSet(value, "C", stamp, stamp + 1); System.out.println("Thread2 将值改为C: " + success + ", 新版本号: " + (stamp + 1)); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } }); // 线程3：再次修改值和版本号 Thread thread3 = new Thread(() -> { try { Thread.sleep(1000); int[] stampHolder = new int[1]; String value = stampedRef.get(stampHolder); int stamp = stampHolder[0]; boolean success = stampedRef.compareAndSet(value, "A", stamp, stamp + 1); System.out.println("Thread3 将值改回A: " + success + ", 新版本号: " + (stamp + 1)); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } }); thread1.start(); thread2.start(); thread3.start(); try { thread1.join(); thread2.join(); thread3.join(); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } public static void main(String[] args) { System.out.println("=== ABA问题演示 ==="); ABAProblemDemo.demonstrateABAProblem(); try { Thread.sleep(3000); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } System.out.println("\n=== ABA问题解决方案 ==="); solveABAProblem(); } }

long startTime = System.nanoTime(); Thread[] threads = new Thread[2]; // 少量线程 for (int i = 0; i < 2; i++) { threads[i] = new Thread(() -> { for (int j = 0; j < OPERATION_COUNT; j++) { atomicCounter.incrementAndGet(); } }); } for (Thread thread : threads) { thread.start(); } for (Thread thread : threads) { try { thread.join(); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } long atomicTime = System.nanoTime() - startTime; System.out.println("CAS性能提升: " + ((double)syncTime / atomicTime) + "倍"); } // 高竞争场景测试 private static void testHighContention() { System.out.println("\n--- 高竞争场景（大量线程，频繁操作） ---"); // AtomicInteger测试 AtomicInteger atomicCounter = new AtomicInteger(0); for (int i = 0; i < THREAD_COUNT; i++) { threads[i] = new Thread(() -> { for (int j = 0; j < OPERATION_COUNT / 10; j++) { syncCounter.increment(); } }); } for (Thread thread : threads) { thread.start(); } for (Thread thread : threads) { try { thread.join(); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } long syncTime = System.nanoTime() - startTime; System.out.println("Synchronized: " + syncTime / 1_000_000 + "ms, 结果: " + syncCounter.get()); if (atomicTime < syncTime) { System.out.println("CAS性能提升: " + ((double)syncTime / atomicTime) + "倍"); } else { System.out.println("Synchronized性能更好: " + ((double)atomicTime / syncTime) + "倍"); } } static class SynchronizedCounter { private int count = 0;
