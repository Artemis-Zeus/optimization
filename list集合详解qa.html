System.out.println("初始容量: 3"); // 添加元素，触发扩容 for (int i = 0; i < 15; i++) { list.add(i); if (i == 2 || i == 4 || i == 7 || i == 11) { System.out.println("添加第" + (i + 1) + "个元素后，可能发生扩容"); } } System.out.println("最终大小: " + list.size()); System.out.println("元素: " + list); } // 3. 自定义ArrayList实现（简化版） public static class SimpleArrayList<E> { private static final int DEFAULT_CAPACITY = 10; private static final Object[] EMPTY_ELEMENTDATA = {}; private Object[] elementData; private int size; // 默认构造器 public SimpleArrayList() { this.elementData = EMPTY_ELEMENTDATA; } // 指定初始容量构造器 public SimpleArrayList(int initialCapacity) { if (initialCapacity > 0) { this.elementData = new Object[initialCapacity]; } else if (initialCapacity == 0) { this.elementData = EMPTY_ELEMENTDATA; } else { throw new IllegalArgumentException("Illegal Capacity: " + initialCapacity); } } // 添加元素 public boolean add(E e) { ensureCapacityInternal(size + 1); elementData[size++] = e; return true; } // 指定位置插入 public void add(int index, E element) { rangeCheckForAdd(index); ensureCapacityInternal(size + 1); // 移动元素 System.arraycopy(elementData, index, elementData, index + 1, size - index); elementData[index] = element; size++; } // 获取元素 public E get(int index) { rangeCheck(index); return (E) elementData[index]; } // 设置元素 public E set(int index, E element) { rangeCheck(index); E oldValue = (E) elementData[index]; elementData[index] = element; return oldValue; } // 删除元素 public E remove(int index) { rangeCheck(index); E oldValue = (E) elementData[index]; int numMoved = size - index - 1; if (numMoved > 0) { System.arraycopy(elementData, index + 1, elementData, index, numMoved); } elementData[--size] = null; // 帮助GC return oldValue; } // 确保容量 private void ensureCapacityInternal(int minCapacity) { if (elementData == EMPTY_ELEMENTDATA) { minCapacity = Math.max(DEFAULT_CAPACITY, minCapacity); } ensureExplicitCapacity(minCapacity); } private void ensureExplicitCapacity(int minCapacity) { if (minCapacity - elementData.length > 0) { grow(minCapacity); } } // 扩容 private void grow(int minCapacity) { int oldCapacity = elementData.length; int newCapacity = oldCapacity + (oldCapacity >> 1); // 扩容1.5倍 if (newCapacity - minCapacity < 0) { newCapacity = minCapacity; } System.out.println("扩容: " + oldCapacity + " -> " + newCapacity); elementData = Arrays.copyOf(elementData, newCapacity); } // 范围检查 private void rangeCheck(int index) { if (index >= size) { throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size); } } private void rangeCheckForAdd(int index) { if (index > size || index < 0) { throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size); } } public int size() { return size; } public boolean isEmpty() { return size == 0; } @Override public String toString() { if (size == 0) return "[]"; StringBuilder sb = new StringBuilder(); sb.append('['); for (int i = 0; i < size; i++) { sb.append(elementData[i]); if (i < size - 1) sb.append(", "); } sb.append(']'); return sb.toString(); } } // 4. 扩容性能测试 public void expansionPerformanceTest() { System.out.println("=== 扩容性能测试 ==="); // LinkedList实现原理分析 public class LinkedListAnalysis { // 1. LinkedList基本操作演示 public void linkedListBasicDemo() { LinkedList<String> list = new LinkedList<>(); System.out.println("=== LinkedList基本操作 ==="); // 添加元素 list.add("元素1"); list.add("元素2"); list.add("元素3"); System.out.println("添加后: " + list); // 头部和尾部操作 list.addFirst("头部元素"); list.addLast("尾部元素"); System.out.println("头尾添加后: " + list); // 队列操作 list.offer("队列元素"); System.out.println("队列添加后: " + list); String first = list.poll(); System.out.println("队列取出: " + first); System.out.println("取出后: " + list); // 栈操作 list.push("栈元素"); System.out.println("栈压入后: " + list); String top = list.pop(); System.out.println("栈弹出: " + top); System.out.println("弹出后: " + list); } // 2. 自定义LinkedList实现（简化版） public static class SimpleLinkedList<E> { private Node<E> first; private Node<E> last; private int size = 0; // 节点类 private static class Node<E> { E item; Node<E> next; Node<E> prev; Node(Node<E> prev, E element, Node<E> next) { this.item = element; this.next = next; this.prev = prev; } @Override public String toString() { return String.valueOf(item); } } // 在尾部添加元素 public boolean add(E e) { linkLast(e); return true; } // 在指定位置添加元素 public void add(int index, E element) { checkPositionIndex(index); if (index == size) { linkLast(element); } else { linkBefore(element, node(index)); } } // 获取元素 public E get(int index) { checkElementIndex(index); return node(index).item; } // 删除元素 public E remove(int index) { checkElementIndex(index); return unlink(node(index)); } // 在尾部链接元素 void linkLast(E e) { final Node<E> l = last; final Node<E> newNode = new Node<>(l, e, null); last = newNode; if (l == null) { first = newNode; } else { l.next = newNode; } size++; } // 在指定节点前插入 void linkBefore(E e, Node<E> succ) { final Node<E> pred = succ.prev; final Node<E> newNode = new Node<>(pred, e, succ); succ.prev = newNode; if (pred == null) { first = newNode; } else { pred.next = newNode; } size++; } // 删除节点 E unlink(Node<E> x) { final E element = x.item; final Node<E> next = x.next; final Node<E> prev = x.prev; if (prev == null) { first = next; } else { prev.next = next; x.prev = null; } if (next == null) { last = prev; } else { next.prev = prev; x.next = null; } x.item = null; size--; return element; } // 根据索引找节点 Node<E> node(int index) { // 优化：根据索引位置选择从头还是从尾开始遍历 if (index < (size >> 1)) { Node<E> x = first; for (int i = 0; i < index; i++) { x = x.next; } return x; } else { Node<E> x = last; for (int i = size - 1; i > index; i--) { x = x.prev; } return x; } } // 检查索引 private void checkElementIndex(int index) { if (index < 0 || index >= size) { throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size); } } private void checkPositionIndex(int index) { if (index < 0 || index > size) { throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size); } } public int size() { return size; } @Override public String toString() { if (size == 0) return "[]"; StringBuilder sb = new StringBuilder(); sb.append('['); Node<E> x = first; while (x != null) { sb.append(x.item); x = x.next; if (x != null) sb.append(", "); } sb.append(']'); return sb.toString(); } } // 3. ArrayList vs LinkedList 性能对比 public void performanceComparison() { final int SIZE = 100000; System.out.println("=== ArrayList vs LinkedList 性能对比 ==="); System.out.println("数据量: " + SIZE); List<Integer> arrayList = new ArrayList<>(); List<Integer> linkedList = new LinkedList<>(); // 1. 尾部添加性能测试 long startTime = System.nanoTime(); for (int i = 0; i < SIZE; i++) { arrayList.add(i); } long arrayListAddTime = System.nanoTime() - startTime; startTime = System.nanoTime(); for (int i = 0; i < SIZE; i++) { linkedList.add(i); } long linkedListAddTime = System.nanoTime() - startTime; System.out.println("尾部添加:"); System.out.println(" ArrayList: " + arrayListAddTime / 1_000_000 + "ms"); System.out.println(" LinkedList: " + linkedListAddTime / 1_000_000 + "ms"); // 2. 随机访问性能测试 Random random = new Random(); startTime = System.nanoTime(); for (int i = 0; i < 10000; i++) { int index = random.nextInt(SIZE); arrayList.get(index); } long arrayListGetTime = System.nanoTime() - startTime; startTime = System.nanoTime(); for (int i = 0; i < 10000; i++) { int index = random.nextInt(SIZE); linkedList.get(index); } long linkedListGetTime = System.nanoTime() - startTime; System.out.println("随机访问(10000次):"); System.out.println(" ArrayList: " + arrayListGetTime / 1_000_000 + "ms"); System.out.println(" LinkedList: " + linkedListGetTime / 1_000_000 + "ms"); // 3. 头部插入性能测试 List<Integer> arrayList2 = new ArrayList<>(); List<Integer> linkedList2 = new LinkedList<>(); startTime = System.nanoTime(); for (int i = 0; i < 10000; i++) { arrayList2.add(0, i); // 头部插入 } long arrayListInsertTime = System.nanoTime() - startTime; startTime = System.nanoTime(); for (int i = 0; i < 10000; i++) { linkedList2.add(0, i); // 头部插入 } long linkedListInsertTime = System.nanoTime() - startTime; System.out.println("头部插入(10000次):"); System.out.println(" ArrayList: " + arrayListInsertTime / 1_000_000 + "ms"); System.out.println(" LinkedList: " + linkedListInsertTime / 1_000_000 + "ms"); } // 4. 使用场景分析 public void usageScenarioAnalysis() { System.out.println("=== ArrayList vs LinkedList 使用场景 ==="); System.out.println("选择ArrayList的场景:"); System.out.println("1. 需要频繁随机访问元素（get操作多）"); System.out.println("2. 主要进行尾部添加操作"); System.out.println("3. 内存使用敏感（ArrayList内存开销更小）"); System.out.println("4. 需要使用索引进行操作"); System.out.println("\n选择LinkedList的场景:"); System.out.println("1. 需要频繁在头部或中间插入/删除元素"); System.out.println("2. 需要实现队列或栈的功能"); System.out.println("3. 不确定数据量大小，避免扩容开销"); System.out.println("4. 很少进行随机访问操作"); // 实际示例：队列操作 System.out.println("\n队列操作示例:"); Queue<String> queue = new LinkedList<>(); queue.offer("任务1"); queue.offer("任务2"); queue.offer("任务3"); while (!queue.isEmpty()) { String task = queue.poll(); System.out.println("处理: " + task); } // 实际示例：栈操作 System.out.println("\n栈操作示例:"); Deque<String> stack = new LinkedList<>(); stack.push("页面1"); stack.push("页面2"); stack.push("页面3"); while (!stack.isEmpty()) { String page = stack.pop(); System.out.println("返回: " + page); } } public static void main(String[] args) { LinkedListAnalysis analysis = new LinkedListAnalysis(); analysis.linkedListBasicDemo(); System.out.println(); // 测试自定义LinkedList System.out.println("=== 自定义LinkedList测试 ==="); SimpleLinkedList<String> customList = new SimpleLinkedList<>(); customList.add("A"); customList.add("B"); customList.add("C"); customList.add(1, "X"); System.out.println("自定义LinkedList: " + customList); System.out.println(); analysis.performanceComparison(); System.out.println(); analysis.usageScenarioAnalysis(); } }
