Q1: OutOfMemoryError有哪些类型？如何定位和解决？

答案：

OOM错误类型分类：

各类型OOM的诊断和解决：

1. Java heap space

# 现象 Exception in thread "main" java.lang.OutOfMemoryError: Java heap space # 排查步骤 jmap -histo:live <pid> # 查看对象统计 jmap -dump:live,format=b,file=heap.hprof <pid> # 生成堆转储 # 解决方案 -Xms4g -Xmx8g # 增加堆内存 -XX:+HeapDumpOnOutOfMemoryError # 自动生成堆转储

2. GC overhead limit exceeded

# 现象 java.lang.OutOfMemoryError: GC overhead limit exceeded # 原因分析 - 98%的时间用于GC，但只回收了不到2%的内存 # 解决方案 -XX:-UseGCOverheadLimit # 关闭GC overhead检查 -XX:+UseG1GC # 使用G1收集器 -Xmx8g # 增加堆内存

3. Direct buffer memory

# 现象 java.lang.OutOfMemoryError: Direct buffer memory # 排查命令 jcmd <pid> VM.native_memory summary # 查看本地内存使用 # 解决方案 -XX:MaxDirectMemorySize=2g # 增加直接内存限制 -Dio.netty.maxDirectMemory=2g # Netty框架特定参数

4. unable to create new native thread

# 现象 java.lang.OutOfMemoryError: unable to create new native thread # 系统检查 ulimit -u # 查看用户进程限制 cat /proc/sys/kernel/threads-max # 查看系统线程限制 # 解决方案 ulimit -u 65536 # 增加用户进程限制 -Xss256k # 减小线程栈大小

Q2: CPU使用率过高如何排查？有哪些常见原因？

答案：

CPU使用率高的常见原因：

CPU问题排查步骤：

# 1. 系统层面分析 top -H -p <pid> # 查看进程线程CPU使用 vmstat 1 5 # 查看系统整体状态 iostat -x 1 5 # 查看IO状态 # 2. Java应用层面分析 jstack <pid> > thread_dump.txt # 生成线程转储 jstat -gc <pid> 1000 # 监控GC状态 jcmd <pid> VM.classloader_stats # 查看类加载状态 # 3. 热点线程分析 # 获取占用CPU最高的线程ID top -H -p <pid> | head -20 # 将线程ID转换为16进制 printf "%x\n" <thread_id> # 在线程转储中查找对应线程 grep -A 20 "nid=0x<hex_thread_id>" thread_dump.txt

CPU问题排查实战案例：

// 问题代码示例 public class CPUHighExample { private volatile boolean running = true; public void problematicMethod() { // 死循环导致CPU飙升 while (running) { // 没有任何sleep或yield，持续占用CPU doSomething(); } } // 修复方案 public void fixedMethod() { while (running) { try { doSomething(); Thread.sleep(10); // 适当休眠 } catch (InterruptedException e) { Thread.currentThread().interrupt(); break; } } } }

自动化CPU分析脚本：

#!/bin/bash # cpu_analysis.sh PID=$1 DUMP_COUNT=${2:-3} INTERVAL=${3:-10} echo "=== CPU Analysis for PID: $PID ===" for i in $(seq 1 $DUMP_COUNT); do echo "=== Dump $i ===" # 获取CPU使用率最高的线程 echo "Top CPU threads:" top -H -p $PID -b -n 1 | head -20 # 生成线程转储 jstack $PID > thread_dump_$i.txt echo "Thread dump saved to thread_dump_$i.txt" if [ $i -lt $DUMP_COUNT ]; then echo "Waiting $INTERVAL seconds..." sleep $INTERVAL fi done echo "Analysis complete. Check thread_dump_*.txt files"

死锁示例代码：

public class DeadlockExample { private final Object lock1 = new Object(); private final Object lock2 = new Object(); public void method1() { synchronized (lock1) { System.out.println("Thread " + Thread.currentThread().getName() + " acquired lock1"); try { Thread.sleep(100); } catch (InterruptedException e) {} synchronized (lock2) { // 可能导致死锁 System.out.println("Thread " + Thread.currentThread().getName() + " acquired lock2"); } } } public void method2() { synchronized (lock2) { System.out.println("Thread " + Thread.currentThread().getName() + " acquired lock2"); try { Thread.sleep(100); } catch (InterruptedException e) {} synchronized (lock1) { // 可能导致死锁 System.out.println("Thread " + Thread.currentThread().getName() + " acquired lock1"); } } } // 解决方案：统一锁顺序 public void fixedMethod1() { synchronized (lock1) { synchronized (lock2) { // 业务逻辑 } } } public void fixedMethod2() { synchronized (lock1) { // 统一先获取lock1 synchronized (lock2) { // 业务逻辑 } } } }

Q7: 如何建立JVM监控和预警体系？

答案：

监控体系架构：

关键监控指标：

# JVM监控指标配置 jvm_metrics: # 内存指标 memory: - heap_used_percent # 堆内存使用率 - heap_committed # 已提交堆内存 - non_heap_used # 非堆内存使用 - metaspace_used # 元空间使用 # GC指标 gc: - gc_count # GC次数 - gc_time # GC耗时 - gc_pause_time # GC停顿时间 - allocation_rate # 内存分配速率 # 线程指标 threads: - thread_count # 线程总数 - blocked_thread_count # 阻塞线程数 - deadlock_count # 死锁数量 # 类加载指标 classloader: - classes_loaded # 已加载类数 - classes_unloaded # 已卸载类数 # 告警规则配置 alert_rules: - name: heap_memory_high condition: heap_used_percent > 85 duration: 5m severity: warning - name: gc_pause_time_high condition: gc_pause_time > 200ms duration: 3m severity: critical - name: thread_deadlock condition: deadlock_count > 0 duration: 1m severity: critical

监控实现示例：

// Spring Boot Actuator + Micrometer @RestController public class HealthController { private final MeterRegistry meterRegistry; private final MemoryMXBean memoryMXBean; private final GarbageCollectorMXBean gcMXBean; @Autowired public HealthController(MeterRegistry meterRegistry) { this.meterRegistry = meterRegistry; this.memoryMXBean = ManagementFactory.getMemoryMXBean(); this.gcMXBean = ManagementFactory.getGarbageCollectorMXBeans().get(0); // 注册自定义指标 Gauge.builder("jvm.memory.heap.used.ratio") .register(meterRegistry, this, this::getHeapUsedRatio); Gauge.builder("jvm.gc.pause.time") .register(meterRegistry, this, this::getLastGCPauseTime); } private double getHeapUsedRatio() { MemoryUsage heapUsage = memoryMXBean.getHeapMemoryUsage(); return (double) heapUsage.getUsed() / heapUsage.getMax(); } private double getLastGCPauseTime() { return gcMXBean.getCollectionTime(); } @GetMapping("/health/jvm") public Map<String, Object> getJVMHealth() { Map<String, Object> health = new HashMap<>(); // 内存信息 MemoryUsage heapUsage = memoryMXBean.getHeapMemoryUsage(); health.put("heap_used_mb", heapUsage.getUsed() / 1024 / 1024); health.put("heap_max_mb", heapUsage.getMax() / 1024 / 1024); health.put("heap_used_percent", getHeapUsedRatio() * 100); // GC信息 health.put("gc_count", gcMXBean.getCollectionCount()); health.put("gc_time_ms", gcMXBean.getCollectionTime()); // 线程信息 ThreadMXBean threadMXBean = ManagementFactory.getThreadMXBean(); health.put("thread_count", threadMXBean.getThreadCount()); health.put("deadlock_count", threadMXBean.findDeadlockedThreads() != null ? threadMXBean.findDeadlockedThreads().length : 0); return health; } }
